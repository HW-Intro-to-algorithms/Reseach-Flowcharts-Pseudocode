START
	int opcion;

		PUTS "3.Rectangulo		4.Ciruculo"
		
		PUTS "Tu seleccion es:	"
		scanf("%d",&opcion)
		switch(opcion){
			float base,altura,areat

				PUTS "Ingrese base del triangulo"
				SCANF "%f",&base

				SCANF "%f",&altura

				arear=baser*alturar;
				PRINT"El area es %f\n",arear
			
			float radio,areac
			CASE 4:
				PUTS "Ingrese radio del circulo"
				SCANF "%f",&radio
